## ğŸ§± Step 2: Filter Component
```jsx
// Filters.js
import React from "react";

function Filters({ filters, onClear, onRemove }) {
  if (filters.length === 0) return null;

  return (
    <div className="filters">
      {filters.map((f, i) => (
        <span key={i}>
          {f}
          <button onClick={() => onRemove(f)}>x</button>
        </span>
      ))}
      <button onClick={onClear}>Clear All</button>
    </div>
  );
}

export default Filters;
```

## ğŸ§± Step 3: Main App Logic
```jsx
// App.js
import React, { useState } from "react";
import jobsData from "./data";
import JobList from "./components/JobList";
import Filters from "./components/Filters";

function App() {
  const [filters, setFilters] = useState([]);

  const handleFilter = (tag) => {
    if (!filters.includes(tag)) {
      setFilters([...filters, tag]);
    }
  };

  const handleRemove = (tag) => {
    setFilters(filters.filter((f) => f !== tag));
  };

  const handleClear = () => setFilters([]);

  const filteredJobs = filters.length === 0
    ? jobsData
    : jobsData.filter((job) =>
        filters.every((f) =>
          job.languages.includes(f) ||
          job.role === f ||
          job.level === f ||
          job.location === f
        )
      );

  return (
    <div className="App">
      <h1>ğŸ’¼ Job Board</h1>
      <Filters filters={filters} onClear={handleClear} onRemove={handleRemove} />
      <JobList jobs={filteredJobs} onFilter={handleFilter} />
    </div>
  );
}

export default App;
```
